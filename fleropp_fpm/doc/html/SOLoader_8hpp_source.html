121 // Type alias for a function pointer to a function that returns a

162 /***
163 * Compares library and source file time modified.
164 *
165 * \return A Bool if both times modified are the different
166 * then return true, otherwise false.
167 */
168 bool was_modified() const
169 {
170 // Source and library file stat structs
171 struct stat src_stat, lib_stat;
172 const auto src_ret = ::stat(_src_path.c_str(), &src_stat) == 0;
173 const auto lib_ret = ::stat(_lib_path.c_str(), &lib_stat) == 0;
174
175 // If library file does not exist, we consider the source file to
176 // have been modified. Otherwise, we return whether the source
177 // is newer than the library based on mtime.
178 if (src_ret && !lib_ret && errno == ENOENT)
179 {
180 return true;
181 }
182 else if (src_ret && lib_ret)
183 {
184 return src_stat.st_mtim.tv_sec > lib_stat.st_mtim.tv_sec;
185 }
186
187 return false;
188 }
