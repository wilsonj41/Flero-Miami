150 void open_lib() final {
151 // Only do something if the library is not currently open
152 static std::shared_mutex open_mtx;
153 if (!m_open.test()) {
154 {
155 // We attempt to acquire an exclusive lock, but don't block if we
156 // cannot acquire it.
157 std::unique_lock lock(open_mtx, std::try_to_lock);
158 if (lock && !m_open.test()) {
159 if ((std::filesystem::file_size(m_shared_object) == 0) ||
160 !(m_handle = ::dlopen(m_shared_object.c_str(), RTLD_LAZY | RTLD_LOCAL))) {
161 spdlog::error("Unable to load DSO '{}': {}", m_shared_object, ::dlerror());
162 } else {
163 m_open.test_and_set();
164 spdlog::debug("Opened '{}'", m_shared_object);
165 }
166 }
167 }
168 }
169 {
170 std::shared_lock lock(open_mtx);
171 }
172 }

184 void close_lib() final {
185 // Only do something if the library is currently open.
186 if (m_open.test()) {
187 // If the handle is valid (i.e., not `nullptr`) *but* we fail to close
188 // it, log an error. Otherwise, we assume the handle was never acquired
189 // in the first place (or freed at some point earlier) and clear the flag.
190 //
191 // The first case also prevents invocation of `::dlclose(nullptr)`, a scenario
192 // that, although not well-documented, is likely to be bad news.
193 if (m_handle && (::dlclose(m_handle) != 0)) {
194 spdlog::error("Unable to unload DSO '{}': {}", m_shared_object, ::dlerror());
195 } else {
196 m_open.clear();
197 spdlog::debug("Closed '{}'", m_shared_object);
198 }
199 }

214 template<typename... Args>
215 std::shared_ptr<T> get_instance(Args&& ...args) {
216 // Type alias for a function pointer to a function that returns a
217 // pointer to `T`
218 using alloc_fun_t = T *(*)(Args&& ...);
219 // Type alias for a function pointer to a void function that
220 // accepts a pointer to `T`
221 using del_fun_t = void (*)(T *);
222
223 // We only need to do the recompile when there is source code
224 // available
225 if (!m_src_path_list.empty()) {
226 // If the library was recompiled, close it so we can reopen
227 recompile();
228 }
229
230 open_lib();
231
232 // dlsym returns a void pointer, so we reinterpret cast it to our
233 // type aliases above
234 auto alloc_fun = reinterpret_cast<alloc_fun_t>(::dlsym(m_handle, m_alloc_sym.c_str()));
235 auto del_fun = reinterpret_cast<del_fun_t>(::dlsym(m_handle, m_del_sym.c_str()));
236
237 // If either is nullptr, something went wrong
238 if (!alloc_fun || !del_fun) {
239 close_lib();
240 spdlog::error("Unable to resolve allocator and/or deleter symbol in '{}': {}", m_shared_object, ::dlerror());
241 // Return a default-constructed std::shared_ptr whose value shall be `nullptr`
242 return {};
243 }
244
245 // Return a shared_ptr to T (whose raw pointer is acquired by
246 // calling alloc_fun) with a custom deleter closure that calls
247 // del_fun.
248 spdlog::debug("Returning instance of '{}'", m_shared_object);
249 return std::shared_ptr<T>{alloc_fun(std::forward<Args>(args)...), [del_fun] (T *p) { del_fun(p); }};
250 }

263 // Checks if the source file was modified
264 bool was_modified() const {
265 // If library file does not exist, we consider the source file to
266 // have been modified. Otherwise, we return whether the source
267 // is newer than the library based on mtime.
268 if (std::filesystem::exists(m_shared_object)) {
269 const auto so_mtim = std::filesystem::last_write_time(m_shared_object);
270 //spdlog::debug("'{}' last modified on {}", m_shared_object, so_mtim);
271 return !m_src_path_list.empty() &&
272 std::any_of(std::begin(m_src_path_list), std::end(m_src_path_list),
273 [so_mtim] (auto src) {
274 return std::filesystem::last_write_time(src) > so_mtim;
275 });
276 }
277 return true;

280 // Compiles the shared object and provides access to the compiler's
281 // exit code and the contents of the stderr stream.
282 std::pair<int, std::string> compile(const std::string& compiler,
283 const std::vector<std::string>& args) const {
284 namespace bp = boost::process;
285 // Set up and execute the compiler
286 // We use boost::process to simplify this process
287 //
288 // Important note: we capture the stderr stream from the compiler
289 // so that it can be logged in the event of a compilation failure.
290 boost::asio::io_context ioc;
291 std::future<std::string> stderr;
292 bp::child chld(bp::search_path(compiler), args, bp::std_in.close(),
293 bp::std_out > bp::null, bp::std_err > stderr, ioc);
294 ioc.run();
295 chld.wait();
296 return {chld.exit_code(), stderr.get()};
297 }

302 // Using source code from the error-generation function provided in the constructor,
303 // compile a dummy shared object that will provide error information in a contextually
304 // relevant manner.
305 void compile_error_object(const std::string& error_text) const {
306 // Generate and stringify a UUID
307 auto tmp_name = "_" + boost::uuids::to_string(boost::uuids::random_generator()());
308
309 // Convert the UUID to a valid C++ identifier
310 tmp_name.erase(std::remove(std::begin(tmp_name), std::end(tmp_name), '-'), std::end(tmp_name));
311
312 // Create a source file name from the UUID and create it within a random tmpdir
313 const auto tmp_file = tmp_name + ".cpp";
314 const auto tmp_dir = boost::filesystem::temp_directory_path() / boost::filesystem::unique_path();
315 const auto tmp_path = tmp_dir / tmp_file;
316 boost::filesystem::create_directories(tmp_dir);
317 std::ofstream fd{tmp_path};
318
319 // Write the desired source code to the temporary file, ready for compilation
320 fd << m_error_generator(tmp_name, error_text, m_shared_object);
321
322 spdlog::info("Compiling dummy error page for '{}'", m_shared_object);
323 const auto [e_exit_code, e_stderr] = compile(m_compiler, {"-std=c++20", "-shared", "-fPIC", "--no-gnu-unique",
324 "-o", m_shared_object, tmp_path.string()});
325
326 // Remove the temporary directory and its contents, as it is no longer needed after compilation
327 boost::filesystem::remove_all(tmp_dir);
328
329 // Last resort; if the supplied error source generation function cannot be compiled,
330 // we fall back to creating an empty dummy file or setting the mtime on the existing file.
331 if (e_exit_code) {
332 spdlog::get("compiler")->warn("Non-zero exit of {} ($? -> {}):\n{}", m_compiler, e_exit_code, e_stderr);
333 // Update the modification time of the existing shared object, if it exists.
334 // If it does not exist, we create an empty file to serve as a
335 // placeholder.
336 //
337 // This signals to `CompUnit` that a recompile has happened---and it
338 // shouldn't happen again until another modification. A failed compile
339 // is, in all likelihood, not going to be resolved until another
340 // modification occurs.
341 if (std::filesystem::exists(m_shared_object)) {
342 // Set the mtime to now
343 const auto new_mtime = std::chrono::system_clock::to_time_t(std::chrono::system_clock::now());
344 boost::filesystem::last_write_time(m_shared_object, new_mtime);
345 spdlog::debug("Updated dummy error page for '{}' modification time to '{}' after failed compile", m_shared_object, new_mtime);
346 } else {
347 // Create an empty file
348 std::ofstream dummy{m_shared_object};
349 spdlog::debug("Created empty placeholder for '{}' after failed compile of dummy error page", m_shared_object);
350 }
351 } else {
352 spdlog::info("Successfully compiled dummy error page for '{}'", m_shared_object);
353 }
354 }

356 // Recompile if necessary
357 void recompile() {
358 static std::shared_mutex recompile_mtx;
359 if (was_modified()) {
360 {
361 // We attempt to acquire an exclusive lock, but don't block if we
362 // cannot acquire it.
363 std::unique_lock lock(recompile_mtx, std::try_to_lock);
364 if (lock && was_modified()) {
365 spdlog::info("Compiling '{}'", m_shared_object);
366 const auto [exit_code, stderr] = compile();
367
368 // If the compilation failed, log the exit code and the errors that were
369 // reported. If it was successful, we can simply close the library.
370 if (exit_code) {
371 spdlog::get("compiler")->warn("Non-zero exit of {} ($? -> {}):\n{}", m_compiler, exit_code, stderr);
372 compile_error_object(stderr);
373 } else { // exit_code == 0 so the page was sucessfully recompiled with no issues.
374 spdlog::info("Sucessfully compiled '{}'", m_shared_object);
375 }
376 close_lib();
377 }
378 }
379 }
380 {
381 std::shared_lock lock(recompile_mtx);
382 }
