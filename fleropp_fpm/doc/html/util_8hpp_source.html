89 template <class Base, class Derived, std::size_t N, std::size_t M, typename... Args>
90 consteval std::array<std::string_view, M> get_overridden(const member_map_t<Base, Derived, N, Args...>& mmap) {
91 // Lambda that determines, given a pair containing a member function
92 // pointer and its textual representation, whether it has been
93 // overridden.
94 const auto is_overridden = [] (const auto& pair) {
95 // If the type of pair.first (the member function) is of the
96 // derived class, then we know it has been overridden.
97 return std::holds_alternative<derived_member_t<Derived, Args...>>(pair.first);
98 };
99
100 // Lambda that, given a pair, returns the second element
101 const auto get_second = [] (const auto& pair) { return pair.second; };
102
103 // Using a view is the easiest way to filter out the names
104 // of member functions that have not been overridden.
105 auto filtered = mmap | std::views::filter(is_overridden)
106 | std::views::transform(get_second);
107
108 // We then copy the resulting contents into a std::array,
109 // as we want to return a container that can be returned
110 // and used outside of the `constexpr` context (e.g., via
111 // assignment to a `static constexpr` variable that may
112 // be printed at runtime). Once c++23 gets better compiler
113 // support, `std::views::to` could do this instead.
114 std::array<std::string_view, M> overridden;
115 std::copy_n(filtered.begin(), M, std::begin(overridden));
